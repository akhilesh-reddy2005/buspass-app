// Firestore Security Rules for User Notification Actions
// Add these rules to your Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Existing rules for other collections should remain...
    
    // User notification actions - allows users to manage their own notification states
        match /userNotificationActions/{actionId} {
          // The document ID format is: "{userId}_{notificationId}"

          // Read a document only if the authenticated user is the owner
          allow read: if request.auth != null
            && request.auth.uid == resource.data.userId;

          // Create only if the authenticated user is the claimed owner in the new data
          allow create: if request.auth != null
            && request.auth.uid == request.resource.data.userId;

          // Update / delete only if the authenticated user matches the existing owner
          allow update, delete: if request.auth != null
            && request.auth.uid == resource.data.userId;
        }
    
    // Original notifications collection - keep existing permissions
    match /notifications/{notificationId} {
      // Users can read their own notifications
      allow read: if request.auth != null && request.auth.uid == resource.data.userId;
      
      // Only admins can create/update/delete notifications
      // You might want to adjust this based on your admin role system
      allow write: if request.auth != null 
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == "admin";
    }
    
    // Users collection - adjust as needed for your role system
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
